SELECT "SBO_BTRUST_LIVE"."OITM"."CodeBars" AS "BarCode",
        "SBO_BTRUST_LIVE"."OITM"."OnHand" as "ItemTotalQty"
        FROM "SBO_BTRUST_LIVE"."PMX_INVT" INNER JOIN "SBO_BTRUST_LIVE"."OITM" ON "PMX_INVT"."ItemCode" = "OITM"."ItemCode"
        inner JOIN "SBO_BTRUST_LIVE"."PMX_ITRI" ON "SBO_BTRUST_LIVE"."PMX_INVT"."ItemTransactionalInfoKey" = "SBO_BTRUST_LIVE"."PMX_ITRI"."InternalKey"
        INNER JOIN "SBO_BTRUST_LIVE"."PDN1" ON "SBO_BTRUST_LIVE"."OITM"."ItemCode" = "SBO_BTRUST_LIVE"."PDN1"."ItemCode"
        INNER JOIN "SBO_BTRUST_LIVE"."OPDN" ON "SBO_BTRUST_LIVE"."PDN1"."DocEntry" = "SBO_BTRUST_LIVE"."OPDN"."DocEntry"
        INNER JOIN  "SBO_BTRUST_LIVE"."PMX_INVD" ON "SBO_BTRUST_LIVE"."OPDN"."ObjType" = "SBO_BTRUST_LIVE"."PMX_INVD"."TransType"
        AND "SBO_BTRUST_LIVE"."PDN1"."DocEntry" = "SBO_BTRUST_LIVE"."PMX_INVD"."DocEntry" AND "SBO_BTRUST_LIVE"."PDN1"."LineNum" = "SBO_BTRUST_LIVE"."PMX_INVD"."DocLineNum"
        AND "SBO_BTRUST_LIVE"."PMX_INVD"."ItemTransactionalInfoKey" = "SBO_BTRUST_LIVE"."PMX_INVT"."ItemTransactionalInfoKey"
        LEFT JOIN "SBO_BTRUST_LIVE"."OBTN" X on X."DistNumber" = "SBO_BTRUST_LIVE"."PMX_ITRI"."BatchNumber" and X."ItemCode" = "SBO_BTRUST_LIVE"."PMX_ITRI"."ItemCode"
        LEFT JOIN "SBO_BTRUST_LIVE"."PMX_ITBA" on "SBO_BTRUST_LIVE"."PMX_ITBA"."BatchAttributeCode" = 'INNER_PACK'
        and "SBO_BTRUST_LIVE"."PMX_ITBA"."ItriKey" = "SBO_BTRUST_LIVE"."PMX_ITRI"."InternalKey"
        Left join "SBO_BTRUST_LIVE"."PMX_ITBA" ITBA_POUNDS on "SBO_BTRUST_LIVE"."PMX_ITBA"."BatchAttributeCode" = 'POUNDS'
        and "SBO_BTRUST_LIVE"."PMX_ITBA"."ItriKey" = "SBO_BTRUST_LIVE"."PMX_ITRI"."InternalKey" Left join "SBO_BTRUST_LIVE"."PMX_ITBA" ITBA_KG
        on "SBO_BTRUST_LIVE"."PMX_ITBA"."BatchAttributeCode" = 'KG' and "SBO_BTRUST_LIVE"."PMX_ITBA"."ItriKey" = "SBO_BTRUST_LIVE"."PMX_ITRI"."InternalKey"
        left join "SBO_BTRUST_LIVE"."OITB" on "SBO_BTRUST_LIVE"."OITB"."ItmsGrpCod" = "SBO_BTRUST_LIVE"."OITM"."ItmsGrpCod"

SELECT 
    OITM."CodeBars" AS "BarCode",
    X."DistNumber" AS "BatchNumber",
    MAX(OITM."OnHand") AS "ItemTotalQty",
    MAX(X."Quantity" - X."QuantOut") AS "BatchQuantity",
    MAX(
        CASE 
            WHEN ITBA_INNER."AttributeValue" IS NULL 
                 OR ITBA_INNER."AttributeValue" = '' 
                 OR TO_BIGINT(ITBA_INNER."AttributeValue") = 0
                 OR X."CostTotal" = 0 
                 OR X."Quantity" = 0
            THEN 0
            ELSE TO_DECIMAL(X."CostTotal"/X."Quantity" * TO_DECIMAL(ITBA_INNER."AttributeValue", 8, 2), 8, 2)
        END
    ) AS "CostPerCase",
    OITM."ItemCode"
FROM SBO_BTRUST_LIVE.PMX_INVT 
INNER JOIN SBO_BTRUST_LIVE.OITM 
    ON PMX_INVT."ItemCode" = OITM."ItemCode" 
INNER JOIN SBO_BTRUST_LIVE.PMX_ITRI 
    ON PMX_INVT."ItemTransactionalInfoKey" = PMX_ITRI."InternalKey" 
INNER JOIN SBO_BTRUST_LIVE.PDN1 
    ON OITM."ItemCode" = PDN1."ItemCode" 
INNER JOIN SBO_BTRUST_LIVE.OPDN 
    ON PDN1."DocEntry" = OPDN."DocEntry" 
INNER JOIN SBO_BTRUST_LIVE.PMX_INVD 
    ON OPDN."ObjType" = PMX_INVD."TransType" 
    AND PDN1."DocEntry" = PMX_INVD."DocEntry" 
    AND PDN1."LineNum" = PMX_INVD."DocLineNum" 
    AND PMX_INVD."ItemTransactionalInfoKey" = PMX_INVT."ItemTransactionalInfoKey" 
LEFT JOIN SBO_BTRUST_LIVE.OBTN X 
    ON X."DistNumber" = PMX_ITRI."BatchNumber" 
    AND X."ItemCode" = PMX_ITRI."ItemCode" 
LEFT JOIN SBO_BTRUST_LIVE.PMX_ITBA ITBA_INNER 
    ON ITBA_INNER."BatchAttributeCode" = 'INNER_PACK' 
    AND ITBA_INNER."ItriKey" = PMX_ITRI."InternalKey" 
LEFT JOIN SBO_BTRUST_LIVE.OITB 
    ON OITB."ItmsGrpCod" = OITM."ItmsGrpCod"
GROUP BY 
    OITM."CodeBars",
    X."DistNumber",
    OITM."ItemCode"
ORDER BY "BarCode"
LIMIT 3000;


SELECT
    T1."ItemCode",
    T0."ItemName",
    T0."FrgnName",
    T0."U_XX_SizeDesc",
    T0."CodeBars",
    CAST(T0."NumInBuy" AS INT) AS "UNIT IN CASE",

    CONCAT(CONCAT(T0."U_XX_SubDept", ' - '), T3."U_XX_SubDept") AS "Sub Dept",
    CONCAT(CONCAT(T0."U_XX_Category", ' - '), T4."U_XX_Description") AS "Category",
    T1."OnHand" / T0."NumInBuy" as "STOCK QTY",
    T0."BuyUnitMsr"
FROM
"SBO_BTRUST_LIVE"."OITM"  T0
INNER JOIN "SBO_BTRUST_LIVE"."OITW"  T1 ON T0."ItemCode" = T1."ItemCode"
LEFT JOIN "SBO_BTRUST_LIVE"."@XX_SUBDEPT"  T3 ON T0."U_XX_SubDept" = T3."Code"
LEFT JOIN "SBO_BTRUST_LIVE"."@XX_CATEGORIES"  T4 ON T0."U_XX_Category" = T4."Code"

--INNER JOIN OBCD T6 ON T0."ItemCode" = T6."ItemCode"
left JOIN "SBO_BTRUST_LIVE"."ITM12" T7 ON T0."ItemCode" = T7."ItemCode" and t7."UomType" = 'I'
left JOIN "SBO_BTRUST_LIVE"."OBCD"  T8 ON t7."BcdEntDft" = T8."BcdEntry"
left JOIN "SBO_BTRUST_LIVE"."ITM12" T9 ON T0."PUoMEntry" = T9."UomEntry"
WHERE
    T1."WhsCode" = 'Main' AND  T1."OnHand" >= '0.01' and  T0."ItemCode" = 'G000915'
GROUP BY
    T1."ItemCode",
    T0."ItemName",
    T0."FrgnName",
    T0."U_XX_SizeDesc",
    T0."CodeBars",
    T0."NumInBuy",
    T0."U_XX_SubDept",
    T3."U_XX_SubDept",
    T0."U_XX_Category",
    T4."U_XX_Description",
    T1."OnHand",
    T0."BuyUnitMsr"
